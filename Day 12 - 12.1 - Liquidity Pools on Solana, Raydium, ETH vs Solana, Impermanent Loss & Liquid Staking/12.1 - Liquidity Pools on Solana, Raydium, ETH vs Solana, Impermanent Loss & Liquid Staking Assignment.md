### Task 3 — Impermanent Loss Calculator

Build a JavaScript function that calculates impermanent loss:

```js
function calculateImpermanentLoss(initialPriceRatio, finalPriceRatio) {
  // priceRatio = price of token A in terms of token B
  // e.g., initialPriceRatio = 1 (1 ETH = 2000 USDC when both sides = $2000)
  // finalPriceRatio = 4 (1 ETH = 8000 USDC)

  const k = finalPriceRatio / initialPriceRatio;

  // Value if held (1 ETH + 2000 USDC, ETH now 4x)
  // Value in pool after rebalancing
  const hodlValue = 1 + k;  // normalized: 1 unit ETH + k units USDC-equivalent
  const poolValue = 2 * Math.sqrt(k); // geometric mean of pool shares

  const impermanentLoss = (poolValue / hodlValue - 1) * 100;

  return {
    priceChange: `${k}x`,
    hodlValue: hodlValue.toFixed(4),
    poolValue: poolValue.toFixed(4),
    impermanentLoss: impermanentLoss.toFixed(4) + '%'
  };
}

// Test cases from lecture:
console.log(calculateImpermanentLoss(1, 4));   // ETH 4x → IL = -20%
console.log(calculateImpermanentLoss(1, 1));   // No change → IL = 0%
console.log(calculateImpermanentLoss(1, 2));   // 2x → IL = -5.7%
console.log(calculateImpermanentLoss(1, 10));  // 10x → IL = -42.5%
```
