    program::invoke_signed,
    system_instruction,
};

pub fn transfer_from_pda(
    program_id: &Pubkey,
    accounts: &[AccountInfo],
    amount: u64,
) -> ProgramResult {
    let accounts_iter = &mut accounts.iter();
    
    let pda_account = next_account_info(accounts_iter)?;
    let destination = next_account_info(accounts_iter)?;
    let system_program = next_account_info(accounts_iter)?;
    
    // TODO: Derive PDA and bump
    let (expected_pda, bump) = Pubkey::find_program_address(
        &[b"vault"],
        program_id
    );
    
    // TODO: Verify PDA
    if pda_account.key != &expected_pda {
        return Err(ProgramError::InvalidAccountData);
    }
    
    // TODO: Create transfer instruction
    let transfer_ix = system_instruction::transfer(
        pda_account.key,
        destination.key,
        amount,
    );
    
    // TODO: Use invoke_signed with correct seeds

        &transfer_ix,
        &[pda_account.clone(), destination.clone(), system_program.clone()],

## Task 7 — Security: Verify PDA Derivation

Add security checks to prevent PDA spoofing.

### Secure Contract

```rust
pub fn process_instruction(
    program_id: &Pubkey,
    accounts: &[AccountInfo],
    _instruction_data: &[u8],
) -> ProgramResult {
    let accounts_iter = &mut accounts.iter();
    let pda_account = next_account_info(accounts_iter)?;
    let user = next_account_info(accounts_iter)?;
    
    // ✅ ALWAYS derive and verify PDA
    let (expected_pda, bump) = Pubkey::find_program_address(
        &[b"vault", user.key.as_ref()],
        program_id
    );
    
    // ✅ Check PDA matches
    if pda_account.key != &expected_pda {
        msg!("Error: Invalid PDA provided");
        msg!("Expected: {}", expected_pda);
        msg!("Received: {}", pda_account.key);
        return Err(ProgramError::InvalidAccountData);
    }
    
    // ✅ Check PDA is owned by this program
    if pda_account.owner != program_id {
        msg!("Error: PDA not owned by program");
        return Err(ProgramError::IncorrectProgramId);
    }
    
    // ✅ Check user is signer
    if !user.is_signer {
        msg!("Error: User must sign");
        return Err(ProgramError::MissingRequiredSignature);
    }
    
    msg!("All security checks passed!");
    Ok(())
}
```

**Requirements:**
- Always verify PDA derivation
- Check account ownership
- Verify signers
- Test with malicious inputs

## Common Mistakes to Avoid

- ❌ **Not verifying PDA derivation**: Always derive and check
- ❌ **Wrong seeds in invoke_signed**: Must match find_program_address
- ❌ **Forgetting bump in signer seeds**: Include `&[bump]` in seeds
- ❌ **Trusting user-provided PDA**: Always derive yourself
- ❌ **Incorrect account ordering**: PDAs must be in right position
- ❌ **Not checking account owner**: Verify PDA is owned by your program
- ❌ **Missing System program**: Include in accounts for create_account
