## 2. Step 1 - Writing the Staking Contract

First, let's create our smart contract in Solidity.

### StakingContract.sol

```solidity
// SPDX-License-Identifier: Unlicense
pragma solidity ^0.8.13;

contract StakingContract {
    uint256 public totalStaked;
    mapping(address => uint256) public stakedBalances;

    function stake(uint256 amount) public payable {
        require(amount > 0, "Amount must be greater than 0");
        require(msg.value == amount, "Amount must be equal to msg.value");
        totalStaked += amount;
        stakedBalances[msg.sender] += amount;
    }

    function unstake(uint256 amount) public payable {
        require(amount <= stakedBalances[msg.sender], "Not enough balance");
        totalStaked -= amount;
        stakedBalances[msg.sender] -= amount;
        payable(msg.sender).transfer(amount);
    }
}
```

### Contract Features

| Function | Description | Visibility |
|----------|-------------|------------|
| `stake(amount)` | Deposit ETH to earn rewards | public, payable |
| `unstake(amount)` | Withdraw staked ETH | public, payable |
| `totalStaked` | Total ETH locked in contract | public view |
| `stakedBalances` | Individual user balances | public view mapping |

### Deploy and Get Contract Address

```bash
# Deploy with Foundry
forge create src/StakingContract.sol:StakingContract \
  --rpc-url $SEPOLIA_RPC \
  --private-key $PRIVATE_KEY

# Note the deployed address
# Example: 0x5BFEaEcf308baB27950a352A79f8aB2795ca1A5A2
```

```typescript
import { sepolia } from 'wagmi/chains'
import { injected } from 'wagmi/connectors'

export const config = createConfig({
  chains: [sepolia],
  connectors: [
    injected(),
  ],
  transports: {
    [sepolia.id]: http(),
  },
})
```

### Configuration Breakdown

```
createConfig()
├── chains: [sepolia]        → Which networks to support
├── connectors: [injected()] → Wallet connection methods (MetaMask)
└── transports: { ... }      → How to connect to blockchain (HTTP RPC)
```

## 6. Step 4 - Building the Appbar (Wallet Connection)

### Create Appbar.tsx

```typescript
import { Connector, useConnect } from 'wagmi'

export function Appbar() {
  const { connectors, connect } = useConnect()

  return (
    <div className="flex justify-between p-2 m-2">
      <div className="text-2xl">
        Stakify
      </div>
      <div>
        {connectors.map((connector) => (
          <button 
            className="mx-2 border rounded p-2" 
            key={connector.uid} 
            onClick={() => connect({ connector })}
          >
            {connector.name}
          </button>
        ))}
      </div>
    </div>
  )
}
```

### How It Works

```
User clicks button
       ↓
connect({ connector }) is called
       ↓
MetaMask popup appears
       ↓
User approves connection
       ↓
App gets wallet address
```

### Conditional Rendering

```typescript
import { useAccount, useConnect, useDisconnect } from 'wagmi'

export function Appbar() {
  const { address } = useAccount()
  const { connectors, connect } = useConnect()
  const { disconnect } = useDisconnect()

  return (
    <div className="flex justify-between p-2 m-2">
      <div className="text-2xl">Stakify</div>
      <div>
        {!address ? (
          // Show connect buttons
          connectors.map((connector) => (
            <button 
              className="mx-2 border rounded p-2" 
              key={connector.uid} 
              onClick={() => connect({ connector })}
            >
              {connector.name}
            </button>
          ))
        ) : (
          // Show connected state
          <div>
            {address ? `Connected: ${address.slice(0,6)}...${address.slice(-4)}` : 'Connect'}
            <button 
              className="mx-2 border rounded p-2" 
              onClick={() => disconnect()}
            >
              Disconnect
            </button>
          </div>
        )}
      </div>
    </div>
  )
}
```

## 7. Step 5 - Building the Dashboard (Stake Button)

### Create Dashboard.tsx

```typescript
import { useWriteContract, useReadContract } from 'wagmi'
import { abi } from './abi'

export function Dashboard() {

  })

  return (
    <div className="h-screen w-screen flex justify-center items-center">
      <div>
        <button 
          className="mx-2 border rounded p-2 text-2xl" 
          onClick={() => {
            writeContract({
              address: '0x5BFEaEcf308baB27950a352A79f8aB2795ca1A5A2',
              abi,
              functionName: 'stake',
              args: [BigInt(1000000000000000000)],
              value: BigInt(1000000000000000000)
            })
          }}
        >
          Stake
        </button>
        <div>
          <ShowStake />
        </div>
      </div>
    </div>
  )
}

function ShowStake() {
  const { data: balance } = useReadContract({
    address: '0x5BFEaEcf308baB27950a352A79f8aB2795ca1A5A2',
    abi,
    functionName: 'stakedBalances',
    args: ['0x2B66473dE390691099F7852009a3709F7d9090d7'],
  })

  return (
    <div>
      You have staked {balance ? (balance.toString()) : '0'} ETH
    </div>
  )
}
```

### Key Hooks Explained

| Hook | Purpose | Returns |
|------|---------|---------|
| `useWriteContract` | Send transactions | `{ writeContract, data: hash }` |
| `useReadContract` | Read contract state | `{ data, isLoading, error }` |
| `useAccount` | Get connected wallet | `{ address, isConnected }` |
| `useConnect` | Connect wallet | `{ connect, connectors }` |

## 8. Complete App.tsx

### Wrap Everything in Providers

```typescript
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { WagmiProvider } from 'wagmi'
import { config } from './config'
import { Appbar } from './Appbar'
import { Dashboard } from './Dashboard'

const queryClient = new QueryClient()

export default function App() {
  return (
    <WagmiProvider config={config}>
      <QueryClientProvider client={queryClient}>
        <Appbar />
        <Dashboard />
      </QueryClientProvider>
    </WagmiProvider>
  )
}
```

### Provider Hierarchy

```
<WagmiProvider>          ← Provides Web3 context
  <QueryClientProvider>  ← Handles data fetching/caching
    <Appbar />           ← Wallet connection UI
    <Dashboard />        ← Staking interface
  </QueryClientProvider>
</WagmiProvider>
```

## 9. Improved Dashboard with Dynamic Address

### Using Connected Wallet Address

```typescript
import { useAccount, useWriteContract, useReadContract } from 'wagmi'
import { abi } from './abi'

const CONTRACT_ADDRESS = '0x5BFEaEcf308baB27950a352A79f8aB2795ca1A5A2'

export function Dashboard() {
  const { address } = useAccount()
  const { writeContract } = useWriteContract()

  if (!address) {
    return (
      <div className="h-screen w-screen flex justify-center items-center">
        <div className="text-2xl">
          Please connect your wallet
        </div>
      </div>
    )
  }

  return (
    <div className="h-screen w-screen flex justify-center items-center">
      <div>
        <button 
          className="mx-2 border rounded p-2 text-2xl" 
          onClick={() => {
            writeContract({
              address: CONTRACT_ADDRESS,
              abi,
              functionName: 'stake',
              args: [BigInt(1000000000000000000)], // 1 ETH
              value: BigInt(1000000000000000000)
            })
          }}
        >
          Stake 1 ETH
        </button>
        <ShowStake userAddress={address} />
      </div>
    </div>
  )
}

function ShowStake({ userAddress }: { userAddress: string }) {
  const { data: balance, isLoading } = useReadContract({
    address: CONTRACT_ADDRESS,
    abi,
    functionName: 'stakedBalances',
    args: [userAddress],
  })

  if (isLoading) return <div>Loading...</div>

  return (
    <div>
      You have staked {balance ? (Number(balance) / 1e18).toFixed(4) : '0'} ETH
    </div>
  )
}
```
